220  // This list of enums is very important! The order in which models are listed
221  // here determines the order of execution of the models.
222  //
223  // There are some conditions that need to be met :
224  // 1. FCS can request mass geometry changes via the inertia/pointmass-*
225  // properties so it must be executed before MassBalance
226  // 2. MassBalance must be executed before Propulsion, Aerodynamics,
227  // GroundReactions, ExternalReactions and BuoyantForces to ensure that
228  // their moments are computed with the updated CG position.
229  enum eModels { ePropagate=0,
230  eInput,
231  eInertial,
232  eAtmosphere,
233  eWinds,
234  eSystems,
235  eMassBalance,
236  eAuxiliary,
237  ePropulsion,
238  eAerodynamics,
239  eGroundReactions,

242  eAircraft,
243  eAccelerations,
244  eOutput,
245  eNumStandardModels };
